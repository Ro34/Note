返回 k 。

class Solution {
    public int removeDuplicates(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }

        int slow = 0;
        for (int fast = 1; fast < nums.length; fast++) {
            if (nums[slow] == nums[fast]) {
                continue;
            } else {
                slow++;
                nums[slow] = nums[fast];

            }

        }
        return slow+1;

    }
}
