class Solution {
    public boolean hasPathSum(TreeNode root, int targetSum) {
        if (root == null) {
            return false;
        }
        targetSum -= root.val;
        if (root.left == null && root.right == null) {
            return targetSum == 0;
        }
        if(root.left != null){

            if(hasPathSum(root.left,targetSum)){
                return true;
            };
//            targetSum += root.left.val;
        }
        if(root.right!= null){

            if(hasPathSum(root.right,targetSum)){
                return true;
            }
//            targetSum -= root.left.val;
        }
        return false;

    }
}
