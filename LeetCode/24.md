class Solution {
    public ListNode swapPairs(ListNode head) {
        if (head == null) {
            return null;
        }
        ListNode curLeft = new ListNode(-1);
        curLeft.next = head;
        ListNode cur = curLeft.next;
        ListNode curRight = curLeft.next.next;
        ListNode temp = null;

        ListNode newHeadNode = head.next;
        while (cur.next != null){
            temp = curRight.next;
            curLeft.next = cur.next;
            curRight.next = cur;
            cur.next = temp;
            curLeft = cur;
        }

        return newHeadNode;


    }
}

class Solution {
    public ListNode swapPairs(ListNode head) {
        if (head == null) {
            return null;
        }
        ListNode dummy = new ListNode(-1);
        dummy.next = head;
        ListNode cur = dummy;
        ListNode leftNode;
        ListNode rightNode;
        ListNode temp;

        while (cur.next != null && cur.next.next != null){
            temp = cur.next.next.next;
            leftNode = cur.next;
            rightNode = cur.next.next;
            cur.next = rightNode;
            rightNode.next = leftNode;
            leftNode.next = temp;
            cur = leftNode;
        }

        return dummy.next;


    }
}
