class Solution {
    public int[] searchRange(int[] nums, int target) {
        int left = 0;
        int right = nums.length - 1;

        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] < target) {
                System.out.println("left+");
                left = mid + 1;
            } else if (nums[mid] > target) {
                System.out.println("right-");
                right = mid - 1;
            } else {
                left = searchLeft(nums, mid, target);
                System.out.println(left);
                right = searchRight(nums, mid, target);
                System.out.println(right);
                return new int[]{left, right};
            }
        }
        return new int[]{-1, -1};
    }

    private int searchRight(int[] nums, int right, int target) {
        while (right <= nums.length - 1) {

            if (nums[right] <= target) {
                right++;
            }
            if (nums[right] > target) {
                return right - 1;
            }
        }
        return -1;
    }

    private int searchLeft(int[] nums, int left, int target) {
        while (left >= 0) {

            if (nums[left] >= target) {
                left--;
            }
            if (nums[left] < target) {
                return left + 1;
            }
        }
        return -1;
    }
}

class Solution {
    public int[] searchRange(int[] nums, int target) {
        if (nums.length == 0) {
            return new int[]{-1, -1};
        }

        int leftIndex = findLeft(nums, target);
        if (leftIndex == -1){
            return new int[] {-1, -1};
        }
        int rightIndex = findRight(nums, target);
        return new int[]{leftIndex,rightIndex};
    }

    public int findLeft (int[] nums, int target){
        int left = 0;
        int right = nums.length - 1;
        while (left < right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] < target) {
                left = mid + 1;
            } else if (nums[mid]> target) {
                right = mid -1;

            }else {
                right = mid;
            }
        }
        if(nums[left] != target){
            return -1;
        }
        return left;
    }

    public int findRight ( int[] nums, int target){
        int left = 0;
        int right = nums.length - 1;
        while (left < right) {
            int mid = left + (right - left+1) / 2;
            if (nums[mid] < target) {
                left = mid + 1;
            } else if (nums[mid]> target) {
                right = mid -1;

            }else {
                left = mid;
            }
        }

        return right;
    }

}
