```
给你单链表的头节点 head ，请你反转链表，并返回反转后的链表。
```

### Me

迭代方法

### mySolution!
```java
class Solution {
    public ListNode reverseList(ListNode head) {
        if (head == null) {
            return null;
        }
        ListNode cur = head;
        ListNode pre = null;
        ListNode temp = null;
        while (cur != null) {
            temp = cur.next;
            cur.next = pre;
            pre = cur;
            cur = temp;
        }
        return pre;

    }
}
```

```java
class Solution {
    public ListNode reverseList(ListNode head) {
        ListNode cur = head;
        ListNode lat = null;
        ListNode pre = null;
        if (head.next == null) {
            return head;
        }
        if (head == null) {
            return null;
        }


        while (cur != null ) {


            lat = cur.next;
            cur.next = pre;
            pre = cur;
            //最后一次迭代，lat为null.  cur->null
            cur = lat;

        }
        return pre;
    }
}
```
