0 <= i, j, k, l < n

class Solution {
    public int fourSumCount(int[] nums1, int[] nums2, int[] nums3, int[] nums4) {
        Map<Integer, Integer> record = new HashMap<Integer, Integer>();
        for (int i : nums1) {
            for (int j : nums2) {
                int sum = i + j;
                record.put(sum,record.getOrDefault(sum,0)+1);

            }
        }
        int res = 0;
        for (int i: nums3) {
           for (int j: nums4) {
               res += record.getOrDefault(0-(i+j),0);
           }
        }
        return res;
    }
}
