class Solution {
    public int evalRPN(String[] tokens) {
        String ch;
        ArrayDeque<Integer> deque = new ArrayDeque<>();
        for (int i = 0; i < tokens.length; i++) {
            ch = tokens[i];
            if ("+".equals(ch)) {
                deque.push(deque.pop() + deque.pop());
            } else if ("-".equals(ch)) {
                deque.push(-deque.pop() + deque.pop());
            } else if ("*".equals(ch)) {
                deque.push(deque.pop() * deque.pop());
            } else if ("/".equals(ch)) {
                int temp1 = deque.pop();
                int temp2 = deque.pop();
                deque.push(temp2 / temp1);
            }else {
                deque.push(Integer.valueOf(ch));
            }
        }
        return deque.pop();
    }
}
